</Boon>

### Solar Ring (Cast) [#solar-ring]

<Boon boon={DbBoons.Apollo.solarRing}>

</Boon>

### Blinding Sprint (Sprint) [#blinding-sprint]

<Boon boon={DbBoons.Apollo.blindingSprint}>

</Boon>

<Boon boon={DbBoons.Apollo.lucidGain}>

</Boon>

### Super Nova

<Boon boon={DbBoons.Apollo.superNova}>

</Boon>

### Light Smite

<Boon boon={DbBoons.Apollo.lightSmite}>

</Boon>

<Boon boon={DbBoons.Apollo.perfectImage}>

</Boon>

### Back Burner

<Boon boon={DbBoons.Apollo.backBurner}>

</Boon>
